\title{Deriving Implication Frames from Large Language Models}
\author{Bradley P.\ Allen\\University of Amsterdam}

\begin{abstract}
This technical note develops formal machinery that operationalizes a central concept of inferentialist semantics, as articulated by Hlobil and Brandom---the range of subjunctive robustness---for use in empirical evaluation of large language models (LLMs). From a large language model $M$ and a set of bearers $B$, the construction derives an implication frame $\langle B, I_M \rangle$ via a selection function on $M$'s bearer expression probabilities, then probes the resulting range of subjunctive robustness as a measure of $M$'s mastery of a given material implication. The frame satisfies Containment by construction, and the standard Hlobil-Brandom constructions apply without modification.
\end{abstract}

\section{Setup}

Let $V$ be a finite set of tokens and $\mathcal{L} = V^*$ the set of finite token sequences. Following Hlobil and Brandom \cite[Def.~61]{hlobil2025}, let $B$ be a non-empty set of \emph{bearers} of implicational roles, abstracted from any particular embodiment. 

The construction below works within the implication-space semantics of Hlobil and Brandom \cite{hlobil2025}, to which the reader is referred for the following machinery used without restatement: the implication frame $\langle B, I \rangle$ with $I \subseteq \wp(B) \times \wp(B)$, $\langle \emptyset, \emptyset \rangle \notin I$, and $\langle B, B \rangle \in I$ \cite[Def.~62]{hlobil2025}; the range of subjunctive robustness $\mathrm{RSR}$ \cite[Def.~63]{hlobil2025}; implication equivalence $\approx$ \cite[Def.~64]{hlobil2025}; implicational role $R(\alpha) = \langle R^+(\alpha), R^-(\alpha) \rangle$ \cite[Def.~65]{hlobil2025}; conceptual content as a pair of roles \cite[Def.~66]{hlobil2025}; the NMMS clauses for the connectives \cite[Def.~70]{hlobil2025}; and Corollary~77, which establishes that a Containment-satisfying frame extended via NMMS yields classical propositional logic in the narrowly logical fragment of the extended consequence relation.

An implication frame $\langle B, I \rangle$ \emph{obeys Containment} iff $\Gamma \cap \Delta \neq \emptyset$ implies $\langle \Gamma, \Delta \rangle \in I$. Containment is the only structural constraint the construction below will impose.

\section{The Derivation}

\begin{definition}[Bearer expression probabilities of an LLM]
\label{def:mu}
Let $M$ be a large language model with completion function $p_M : \mathcal{L} \to \Delta(\mathcal{L})$ assigning to each context $c \in \mathcal{L}$ a probability distribution over token sequences, and let $\mathrm{Expr} \subseteq \mathcal{L} \times B$ be an expression relation such that $\langle \sigma, \varphi \rangle \in \mathrm{Expr}$ iff $\sigma$ expresses $\varphi$. For each $\varphi \in B$, let
\[
E_\varphi = \{\sigma \in \mathcal{L} : (\sigma, \varphi) \in \mathrm{Expr}\}
\]
be the event that a completion expresses $\varphi$. The \emph{bearer expression probabilities of $M$} are given by the function $\mu_M : \mathcal{L} \times B \to [0,1]$ where
\[
\mu_M(c)(\varphi) \;=\; p_M(c)(E_\varphi) \;=\; \sum_{\sigma \in E_\varphi} p_M(c)(\sigma).
\]
That is, $\mu_M(c)(\varphi)$ is the probability under $p_M(c)$ that $M$'s completion of $c$ expresses $\varphi$.
\end{definition}

\begin{remark}[Expression events]
\label{rem:expression-events}
Each $\mu_M(c)(\varphi)$ is the probability of a separate event in $\mathcal{L}$ under $p_M(c)$. The events $E_\varphi$ may overlap (a single $\sigma$ may express several bearers) and need not cover $\mathcal{L}$ (some $\sigma$ may express no bearer in $B$), so $\sum_{\varphi \in B} \mu_M(c)(\varphi)$ may exceed or fall short of $1$.
\end{remark}

For $\Gamma \subseteq B$, let $\mathrm{Ctx}_\Gamma = \{c \in \mathcal{L} : (c, \gamma) \in \mathrm{Expr} \text{ for all } \gamma \in \Gamma\}$ denote the set of contexts that express every bearer in $\Gamma$. A \emph{selection function} is a map $S$ that takes a function $f : B \to [0,1]$ to a subset $S(f) \subseteq B$. Fix throughout an admissible selection function $S$; we write $I_M$ for the frame so derived.

\begin{definition}[Derived implication frame]
\label{def:frame}
Given an LLM $M$ with bearer expression probabilities $\mu_M$ and a selection function $S$, the \emph{derived implication frame for $M$} is $\langle B, I_M \rangle$ where $\langle \Gamma, \Delta \rangle \in I_M$ iff at least one of the following holds:
\begin{enumerate}[label=(\roman*)]
\item $\Gamma \cap \Delta \neq \emptyset$; or
\item $\exists c \in \mathrm{Ctx}_\Gamma$ such that $\Delta \cap S(\mu_M(c)) \neq \emptyset$.
\end{enumerate}
\end{definition}

\begin{remark}[Existential by design]
Clause (ii) of Definition~\ref{def:frame} is existential over $c \in \mathit{Ctx}_\Gamma$: $\langle \Gamma, \Delta \rangle \in I_M$ iff \emph{some} premise-expressing context yields $\Delta$-mass under $S$. A universal variant---requiring $\Delta$-mass under every $c \in \mathit{Ctx}_\Gamma$, or every $c$ in some specified subset---is formally available and would yield a tighter frame $I_M^\forall \subseteq I_M$. The two correspond to different conceptions of what it is for $M$ to endorse an inference: clause (ii) treats endorsement as the \emph{availability} of inferential disposition under some appropriate framing, while a universal variant would treat endorsement as the \emph{uniformity} of that disposition across framings. The construction takes no position on which conception is correct; both yield Containment-satisfying frames and both admit the downstream Hlobil--Brandom constructions. 
% The choice between them interacts with the brittleness controls discussed in Section~\ref{sec:probing}.
\end{remark}

\begin{definition}[Selection functions]
\label{def:selection}
The construction admits a range of selection functions; three natural instances are:
\begin{itemize}
\item \emph{Top-$k$}: for $k \geq 1$, $\mathrm{Top}_k(\mu_M(c))$ is the set of bearers $\varphi$ with $\mu_M(c)(\varphi) > 0$ among the $k$ with largest $\mu_M(c)(\varphi)$, ties broken by a fixed convention on $B$.
\item \emph{Absolute threshold}: for $\tau \in (0, 1]$, $\mathrm{Thr}_\tau(\mu_M(c)) = \{\varphi \in B : \mu_M(c)(\varphi) > \tau\}$.
\item \emph{Nucleus} (top-$p$): for $p \in (0, 1]$, ordering bearers $\varphi_1, \varphi_2, \ldots$ by descending $\mu_M(c)(\varphi)$ with ties broken by a fixed convention, $\mathrm{Nuc}_p(\mu_M(c)) = \{\varphi_1, \ldots, \varphi_n\}$ where $n$ is the smallest index with $\sum_{i=1}^n \mu_M(c)(\varphi_i) \geq p$, or all bearers with positive mass if no such $n$ exists. Cumulative mass is taken in absolute terms; by Remark~\ref{rem:expression-events}, $\mu_M(c)$ is not a pmf on $B$.
\end{itemize}
\end{definition}

Clause (ii) of Definition~\ref{def:frame} is non-trivial under any of these instances: it requires positive support from $M$ for some $\delta \in \Delta$, strictly stronger than ``the completion is possible'' (vacuous under positive-temperature sampling) and, when $S(\mu_M(c))$ is non-singleton, strictly weaker than ``the completion is the single most probable output''.

\begin{remark}[Methodological commitment of $S$]
\label{rem:methodological}
The choice of selection function is a substantive methodological commitment that falls to the analyst. Top-$k$ is engineering convenience: $k$ is chosen for tractability and has no independent principled basis. Absolute threshold operationalizes the Lockean thesis from formal epistemology---that credence above a fixed threshold counts as commitment \cite{foley1992}---and inherits its known difficulties (the lottery and preface paradoxes). Nucleus matches the regime under which deployed LLMs generate text \cite{holtzman2019}, but adapts cardinality to context rather than answering to an independent epistemic norm. The downstream structural properties of $I_M$, and the content structure derived from it, depend on $S$.
\end{remark}

\begin{remark}[Bearers in the construction]
\label{rem:bearersassentences}
In the construction developed here, bearers are taken to be propositional contents, with the expression relation $\mathit{Expr}$ pairing token sequences that express these contents. The examples below illustrate this choice.
\end{remark}

\begin{example}[Capital-of-France]
\label{ex:capital}
Take $c = {}$``The capital of France is'' and let $B$ contain
\begin{align*}
\varphi_1 &= \text{``Paris is the capital of France''},\\
\varphi_2 &= \text{``Lyon is the capital of France''}.
\end{align*}
Suppose $p_M(c)$ assigns positive mass to several completions:
\begin{center}
\begin{tabular}{lcc}
\hline
$\sigma$ & $p_M(c)(\sigma)$ & expresses\\
\hline
``Paris.'' & $0.60$ & $\varphi_1$\\
``Paris, of course.'' & $0.15$ & $\varphi_1$\\
``the city of Paris.'' & $0.05$ & $\varphi_1$\\
``Lyon.'' & $0.02$ & $\varphi_2$\\
``a beautiful country.'' & $0.10$ & ---\\
(other) & $0.08$ & ---\\
\hline
\end{tabular}
\end{center}
Then $\mu_M(c)(\varphi_1) = 0.80$ and $\mu_M(c)(\varphi_2) = 0.02$. Two things are visible: (a) $\mathrm{Expr}$ collapses three surface forms into one bearer-level number; (b) $\mu_M(c)(\varphi_1) + \mu_M(c)(\varphi_2) = 0.82 < 1$, since the remaining mass lands outside $E_{\varphi_1} \cup E_{\varphi_2}$ (cf.\ Remark~\ref{rem:expression-events}).

Under top-$1$, $S(\mu_M(c)) = \{\varphi_1\}$. Under absolute threshold with $\tau = 0.5$, also $\{\varphi_1\}$. Under nucleus with $p = 0.7$, also $\{\varphi_1\}$ (the singleton already meets the cumulative-mass threshold). All three agree here; for a more diffuse $p_M(c)$ they would diverge. By clause (ii) of Definition~\ref{def:frame}, $\langle \Gamma, \{\varphi_1\} \rangle \in I_M$ for any $\Gamma$ with $c \in \mathrm{Ctx}_\Gamma$.
\end{example}

\begin{example}[Stop sign and the range of subjunctive robustness]
\label{ex:stopsign}
Following Simonelli's discussion of GPT-4.1 \cite{simonelli2026}, take $B$ to contain $s_a = {}$``$a$ is a stop sign'', $r_a = {}$``$a$ is red'', $n = {}$``it is nighttime'', and $b_a = {}$``$a$ has been painted blue''. Three of Simonelli's prompts witness three premise sets:
\begin{center}
\begin{tabular}{lll}
\hline
$\Gamma$ & $c \in \mathrm{Ctx}_\Gamma$ & top completion expresses $r_a$?\\
\hline
$\{s_a\}$ & ``Are stop signs red?'' & yes\\
$\{s_a, n\}$ & ``Is a stop sign at night still red?'' & yes\\
$\{s_a, b_a\}$ & ``Is a stop sign that's been painted blue still red?'' & no\\
\hline
\end{tabular}
\end{center}
$M$'s top completion thus tracks $\{r_a\}$ under the $\{s_a\}$ and $\{s_a, n\}$ framings but not under $\{s_a, b_a\}$. Under top-$1$, clause (ii) of Definition~\ref{def:frame} delivers
\[
\langle\{s_a\}, \{r_a\}\rangle \in I_M, \quad \langle\{s_a, n\}, \{r_a\}\rangle \in I_M, \quad \langle\{s_a, b_a\}, \{r_a\}\rangle \notin I_M
\]
(the last is not rescued by clause (i), since $\{s_a, b_a\} \cap \{r_a\} = \emptyset$). Applying the Hlobil--Brandom range-of-subjunctive-robustness construction \cite[Def.~63]{hlobil2025} then yields
\[
\{n\} \in \mathrm{RSR}(\langle\{s_a\}, \{r_a\}\rangle), \qquad \{b_a\} \notin \mathrm{RSR}(\langle\{s_a\}, \{r_a\}\rangle).
\]
This is the defeasibility pattern Simonelli identifies as the mark of GPT-4.1's mastery of the material inferential rule $s_a \vdash r_a$; the construction turns the anecdotal evidence into a definite procedure for probing the RSR.
\end{example}

\begin{remark}[Stratification]
\label{rem:stratification}
Definition~\ref{def:frame} does not refer to $R$, $\approx$, or content. The content structure that arises from applying the Hlobil--Brandom construction to $I_M$ is therefore a consequence of the derivation, not a presupposition. Circularity that would arise from defining $I_M$ via content-equivalence of contexts is blocked at the bearer level.
\end{remark}

\begin{remark}[Containment by construction]
\label{rem:containment}
By clause (i) of Definition~\ref{def:frame}, $\langle B, I_M \rangle$ obeys Containment.
\end{remark}

\section{Discussion}
\label{sec:discussion}

\begin{remark}[Stance on measurement]
The construction does not identify $I_M$ with $M$'s content. The claim is rather that $I_M$ is the inferentialist-semantic characterization of $M$'s behavior under content-individuation supplied by the analyst---the bearer set $B$, the expression relation $\mathit{Expr}$, and the selection function $S$. The stratification noted in Remark~\ref{rem:stratification} ensures that the downstream content structure ($R$, $\approx$, conceptual content) is derived from $I_M$ rather than presupposed in its construction, but it does not eliminate the analyst's role in fixing the inputs. Whether the resulting characterization captures $M$'s content in any stronger sense---whether, for instance, $M$ itself individuates the bearers of $B$, or grasps the role-pairs that emerge from $I_M$---is a question this construction does not address. The construction provides a determinate object for inferentialist-semantic analysis of $M$; it does not settle the further question of what relationship that object bears to $M$'s content as such.
\end{remark}

With $\langle B, I_M \rangle$ in hand, the standard Hlobil--Brandom machinery applies without modification: $\mathrm{RSR}$, $\approx$, implicational roles, conceptual contents, and extensions of the material base with logical vocabulary are all obtained by their constructions operating on $\langle B, I_M \rangle$ as an ordinary implication frame. Nothing specific to the LLM origin of $I_M$ enters these constructions.

\begin{remark}[Classical core]
By Remark~\ref{rem:containment}, $\langle B, I_M \rangle$ satisfies Containment, so Corollary~77 of Hlobil and Brandom~\cite{hlobil2025} applies: the narrowly logical fragment of the consequence relation obtained by extending $\langle B, I_M \rangle$ via the NMMS clauses is classical propositional logic. This holds for any $M$ and any admissible parameters; it is a property of the NMMS-extended consequence relation, not of $I_M$. The material base may be substructural while the extension carries a classical logical core: whether $I_M$ is in fact nonmonotonic, nontransitive, or non-contractive is an empirical property of $\mu_M$, not something the construction forces or forbids.
\end{remark}

Implicational role inclusion and content inclusion \cite[Def.~90, 93]{hlobil2025} are likewise available in the derived frame: one can ask which bearers $M$ treats as intersubstitutable \emph{salva consequentia} and which contents are included in which others. These questions have determinate answers given $\langle B, I_M \rangle$ and characterize aspects of $M$'s behavior not directly visible in $\mu_M$.

The construction thus provides a determinate inferentialist-semantic characterization of $M$: given $M$, a bearer set $B$, and admissible parameters, the content structure derived from $\langle B, I_M \rangle$ is well-defined and characterizes $M$'s material-inferential dispositions at the level of role and content identity. In addition, the construction yields not just a frame but a definite probing procedure. 

\begin{definition}
Let $\langle X, A \rangle$ be a target implication and $\mathcal{Y} \subseteq \wp(B)$ be a finite \emph{probe set} of candidate side premises. For each $Y \in \mathcal{Y}$, the \emph{probed verdict} is the clause-(ii) outcome:
\[
\rho_M^S(Y) \;=\; 1 \iff \exists c \in \mathrm{Ctx}_{X \cup Y}, \; A \cap S(\mu_M(c)) \neq \emptyset.
\]
Equivalently, $\rho_M^S(Y) = 1$ iff $Y \in \mathrm{RSR}(\langle X, A \rangle)$ as derived from $I_M$. The function $\rho_M^S$ probes the RSR at the points specified by $\mathcal{Y}$; it does not enumerate it (the RSR is in general countably infinite). 
\end{definition}

\begin{remark}[Probing versus defining]
Note that $\rho^S_M$ as deployed evaluates the existential at a particular $c$ rather than searching $\mathrm{Ctx}_{X \cup Y}$; a positive verdict witnesses $I_M$-membership, while a negative verdict reflects the chosen $c$ rather than ranging over the existential. The relationship between $\rho^S_M$ under a given choice of $c$ and the frame $I_M$ as defined is itself a question this note leaves open. 
\end{remark}

\begin{remark}[Sensitivity to inputs]
The construction depends on analyst-supplied inputs at several points: the bearer set $B$, the expression relation $\mathrm{Expr}$, the selection function $S$, and (where the construction is applied to probe the RSR) a reference standard against which $M$'s verdicts are compared. The resulting frame $\langle B, I_M \rangle$, and the content structure derived from it via the standard Hlobil--Brandom constructions, are sensitive to these choices. Different bearer sets carve $M$'s behavior at different joints; different selection functions yield tighter or more permissive frames; different reference standards register $M$'s agreement with different intended verdict patterns. The construction provides a determinate object for inferentialist-semantic analysis given the inputs, but does not adjudicate which inputs are correct.
\end{remark}
